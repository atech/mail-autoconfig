def client_config
  return @client_config if @client_config
  
  @client_config = MailAutoconfig::ClientConfig.search(domain)
  unless @client_config
    @client_config = MailAutoconfig::ClientConfig.search(primary_mx_domain)
  end
  
  @client_config
end
